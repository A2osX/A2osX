NEW
  AUTO 3,1
*--------------------------------------
* OUT:
*  CS = Error
*  CC/VC : Sst(Y) is not a PIPE
*  CC/VS : Sst(Y) is a PIPE
*--------------------------------------
X.IsPipeIn		ldy #S.PS.pStdIn
*--------------------------------------
* IN:
*  Y = S.PS.pStdOut ...
*--------------------------------------
X.IsPipeY		jsr A2osX.GetPSy
				pha
				iny
				jsr A2osX.GetPSy
				ply

				>STYA ZPPtr1

				>LIBC FileNo
				bcs .99

				>SS
				>PUSHYA
				>PUSHEA.G STATBUF
				>LIBC FStat
				>SR
				bcs .99

				>LDA.G STATBUF+S.STAT.MODE+1
				and #$F0

				cmp /S.STAT.MODE.FIFO
				clv						Clear V
				bne .8

				>LDYA ZPPtr1

				bit .99					Set V

.8				clc
				rts

.98				sec
.99				rts
*--------------------------------------
