NEW
  PREFIX
AUTO 4,1
#!/BIN/SH
#
#  If Level test
#
READ -P "Enter a number 1 through 7:  " ANum
IF [ $ANum = "1" ]
  ECHO "1 Typed"
ELSE
  IF [ $ANum = "2" ]
    ECHO "2 Typed"
  ELSE
    IF [ $ANum = "3" ]
      ECHO "3 Typed"
    ELSE
      IF [ $ANum = "4" ]
        ECHO "4 Typed"
      ELSE
        IF [ $ANum = "5" ]
          ECHO "5 Typed"
        ELSE
          IF [ $ANum = "6" ]
            ECHO "6 Typed"
          ELSE
            IF [ $ANum = "7" ]
              ECHO "7 Typed"
            ELSE
              ECHO "Bad Choice Typed"
            FI
          FI
        FI
      FI
    FI
  FI
FI
MAN
TEXT /MAKE/USR/SHARE/TESTS/IFLEVEL
