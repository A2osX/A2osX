NEW
  AUTO 3,1
				.LIST OFF
				.OP 65C02
				.OR $2000
				.TF root/sha256test
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/a2osx.api.i
				.INB inc/kernel.i
				.INB inc/libcrypto.i
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
ZPPtrC          .BS 2
ZPPtrD			.BS 2
ZPCtx           .BS 2
ZS.END          .ED
*--------------------------------------
CS.START        cld
				jmp (.1,x)
				.DA #$61			; 6502,Level 1 (65c02)
				.DA #2				; BIN Layout Version 2
				.DA #0
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64
				.DA #ZS.END-ZS.START
*--------------------------------------
.1              .DA CS.INIT
				.DA CS.RUN
				.DA CS.DOEVENT
				.DA CS.QUIT
L.libcrypto		.DA libcrypto
L.MSG.TITLE		.DA MSG.TITLE
L.MSG.LOADFAIL	.DA MSG.LOADFAIL
L.MSG.T1		.DA MSG.T1
L.TV1.MSG		.DA TV1.MSG
L.TV1.EXPECTED	.DA TV1.EXPECTED
L.MSG.T2		.DA MSG.T2
L.TV2.MSG		.DA TV2.MSG
L.TV2.EXPECTED	.DA TV2.EXPECTED
L.MSG.T3		.DA MSG.T3
L.TV3.MSG		.DA TV3.MSG
L.TV3.EXPECTED	.DA TV3.EXPECTED
L.MSG.GOT		.DA MSG.GOT
L.MSG.PASS		.DA MSG.PASS
L.MSG.FAIL		.DA MSG.FAIL
L.MSG.LIBERR	.DA MSG.LIBERR
L.MSG.DONE		.DA MSG.DONE
L.FMT.HEX16		.DA FMT.HEX16
				.DA 0
*--------------------------------------
CS.INIT
CS.QUIT         clc
				rts
*--------------------------------------
CS.RUN
				>SS
				>LDYA L.MSG.TITLE
				>LIBC PutS
				>SR
				
				>LDYA L.libcrypto
				>KAPI LoadLib
				bcc .1
				>LDYA L.MSG.LOADFAIL
				>LIBC PutS
				sec
				rts

.1				>STYA LIBCRYPTO

				; Point ZPPtrD at DigestBuf in DS segment
				>LEA.G DigestBuf
				>STYA ZPPtrD

				* Test 1 - empty string
				>SS
				>LDYA L.MSG.T1
				>LIBC PutS
				>SR

				>PUSHW L.TV1.MSG
				>PUSHWI 0
				>PUSHW ZPPtrD
				>LIBCALL LIBCRYPTO,SHA256
				bcc ok1
                >LDYA L.MSG.LIBERR
				>LIBC PutS
				sec
				rts
				

ok1				>LDYA L.TV1.EXPECTED
				>STYA ZPPtrC
				jsr PrintAndCompare

				* Test 2 - "abc"
				>SS
				>LDYA L.MSG.T2
				>LIBC PutS
				>SR

				>LDYA L.TV2.MSG
				>PUSHYA
				>PUSHWI 3
				>PUSHW ZPPtrD
				>LIBCALL LIBCRYPTO,SHA256
				bcc ok2
                >LDYA L.MSG.LIBERR
				>LIBC PutS
				sec
				rts

ok2				>LDYA L.TV2.EXPECTED
				>STYA ZPPtrC
				jsr PrintAndCompare

				* Test 3 - 56 byte NIST vector
				>SS
				>LDYA L.MSG.T3
				>LIBC PutS
				>SR

				>LDYA L.TV3.MSG
				>PUSHYA
				>PUSHWI 56
				>PUSHW ZPPtrD
				>LIBCALL LIBCRYPTO,SHA256
				bcc ok3
                >LDYA L.MSG.LIBERR
				>LIBC PutS
				sec
				rts

ok3				>LDYA L.TV3.EXPECTED
				>STYA ZPPtrC
				jsr PrintAndCompare

				>SS
				>LDYA L.MSG.DONE
				>LIBC PutS
				>SR

				>LDYA LIBCRYPTO
				>KAPI UnloadLib

				lda #0
				sec
				rts

*--------------------------------------
* PrintAndCompare
* ZPPtrC = expected digest (32 bytes)
* ZPPtrD = computed digest
*--------------------------------------
PrintAndCompare
				>LDYA L.MSG.GOT
				>LIBC PutS

				>SS
				>PUSHW L.FMT.HEX16
				ldy #0
.1				lda (ZPPtrD),y
				>PUSHA
				iny
				cpy #16
				bne .1
				>PUSHBI 16
				>LIBC PrintF
				>SR
				>SS
				>PUSHW L.FMT.HEX16
				ldy #16
.2				lda (ZPPtrD),y
				>PUSHA
				iny
				cpy #32
				bne .2
				>PUSHBI 16
				>LIBC PrintF
				>SR

				* Compare
				ldy #31
.3				lda (ZPPtrD),y
				cmp (ZPPtrC),y
				bne fail
				dey
				bpl .3
				
				>SS
				>LDYA L.MSG.PASS
				>LIBC PutS
				>SR
				rts

fail			>SS
				>LDYA L.MSG.FAIL
				>LIBC PutS
				>SR
				rts
*--------------------------------------
CS.DOEVENT      sec
				rts
*--------------------------------------
CS.END          .EQ *
*--------------------------------------
libcrypto       .AZ "libcrypto"

FMT.HEX16       .CZ "%h%h%h%h %h%h%h%h %h%h%h%h %h%h%h%h \r\n"

MSG.TITLE       .AZ "SHA256 Test Vectors"
MSG.T1          .AZ "Test 1: SHA256(null)"
MSG.T2          .AZ "Test 2: SHA256(abc)"
MSG.T3          .AZ "Test 3: SHA256(448-bit)"
MSG.GOT         .AZ "  Computed:"
MSG.PASS        .AZ "  PASS"
MSG.FAIL        .AZ "  FAIL"
MSG.DONE        .AZ "All tests complete"
MSG.LIBERR      .AZ "Call failed"
MSG.LOADFAIL    .AZ "Failed to load libcrypto"

TV1.MSG         .AZ ""
TV2.MSG         .AZ "abc"
TV3.MSG         .AZ "abcdbcdecdefdefgefghfghighijhijkijkljklmklmnlmnomnopnopq"

TV3.EXPECTED
				.HS 248D6A61D20638B8

				.HS A33CE45964FF2167
				.HS F6ECEDD419DB06C1

*--------------------------------------
ID.END          .EQ *
*--------------------------------------
				.DUMMY
				.OR 0
DS.START
DigestBuf       .BS 32
DS.END          .ED
