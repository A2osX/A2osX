# Title Box Function
# CALL TBOX Str_Title
FUNCTION TBOX {
  ECHO -N "\e[1;20H"
  ECHO "\e[7m                                        \e[0m"
  SET LINE = 2
  WHILE [ $LINE -NE 5 ]
    ECHO -N "\e[${LINE};20H"
    ECHO "\e[7m \e[0m                                      \e[7m \e[0m"
    SET LINE = $LINE + 1
  LOOP
  ECHO -N "\e[5;20H"
  ECHO "\e[7m                                        \e[0m"
  ECHO -N "\e[3;22H $1"
  CALL PRINTXY 7 0 ""
}
# Print Inverse
FUNCTION INVERSE {

}

# CALL PRINTXY Num_Row Num_Column Str_String

# CALL INVERSEXY Row Col String
