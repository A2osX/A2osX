				.FIN

				.EM

				.EM

				.EM

				.EM

				.MA PUSHB

				lda ]1,]2
				>PUSHA
				.ELSE
				lda ]1
				>PUSHA
				.FIN
				.EM
*--------------------------------------
				.MA PULLB
				.DO ]#=2
				>PULLA

				.ELSE

				sta ]1
				.FIN
				.EM

				.MA PUSHWZ

				>PUSHA

				.EM

				>PUSHA

				>PUSHA
				.EM

				lda	]1+1,]2
				>PUSHA

				>PUSHA
				lda ]1
				>PUSHA
				.FIN
				.EM

				.MA PULLW
				>PULLA

				.EM

				.EM

				bne :1

*--------------------------------------
				.MA POP
				.DO ]1=1
				inc pStack
				.FIN
				.DO ]1=2
				inc pStack
				inc pStack
				.FIN
				.DO ]1=3
				inc pStack
				inc pStack
				inc pStack
				.FIN
				.DO ]1>3
				pha
				lda #]1
:1				inc pStack
				dec
				bne :1
				pla
				.FIN
				.EM
