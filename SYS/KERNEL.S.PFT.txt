PFT.GetPath0	ldy #0
				.HS 2C					BIT ABS
PFT.GetPath1	ldy #1
				.HS 2C					BIT ABS
PFT.GetPath2	ldy #2
				.HS 2C					BIT ABS
PFT.GetPath4	ldy #4

				jsr RAMSW.GetStkYW

PFT.GetPathYA	jsr RAMSW.YA2PathBuf

				jsr STDLIB.Expand
				jmp STDLIB.RealPath
*--------------------------------------
* In:
*  K.MLI.P1
* Out:
* VC
*  X=FSID, K.MLI.P1= L+prodosvol, L+prodosvol/path, L+mount or L+mount/path
* VS
*  X = 0 /
*  X = 2 /mnt
*  X = 4 /dev
*--------------------------------------
PFT.Translate	>LDYAI K.MLI.P1
				>STYA K.MLI.PARAMS+1

				stz K.MLI.P1

				jsr PFT.CheckVDirs
				bcc .20					/mnt /dev ....

				ldy #$ff

				ldx #0

.10				iny
				lda SYS.BOOT,y			add /BOOT/DIR/ prefix
				beq .11

				sta K.MLI.P1+1,x
				inx
				cpx #MLI.MAXPATH+1
				bcc .10

				lda #MLI.E.INVPATH
*				sec
				rts
*--------------------------------------
.11				stx K.MLI.P1

				ldy #1					skip leading /
				jsr PFT.Y2MLIPath
				bcs .9

				ldx #0

				clv						not virtual DIR

*				clc

				rts
*--------------------------------------
.20				bvs	.30					CC/VS, /mnt or /dev, X = virtual DIR

				phx

				ldy #4					CC/VC, skip /mnt /dev
				jsr PFT.Y2MLIPath

				plx

				bcs .9

*				clv

				rts
*--------------------------------------
.30

*				clc

.9				rts
*--------------------------------------
* In:
*  K.PathBuf
* Out:
*  CC,VC: /dev/file or /mnt/dir
*  CC,VS: /dev or /mnt
*  CS: not a virtual dir
*--------------------------------------
PFT.CheckVDirs	ldx #4

.1				lda PFT.Dirs-2,x
				sta .3+1
				lda PFT.Dirs-1,x
				sta .3+2

				ldy #0
				bra .3

.2				cmp K.PathBuf,y
				bne .7

				iny

.3				lda $FFFF,y				SELF MODIFIED
				bne .2

				lda K.PathBuf,y
				bne .4

				lda #'/'				/XXX\0, append '/'
				sta K.PathBuf,y
				iny
				lda #0
				sta K.PathBuf,y

.8				bit .9					set V

				clc

				rts

.4				cmp #'/'
				bne .7

				iny
				lda K.PathBuf,y
				beq .8					/XXX/\0, V set

				clv						/XXX/something.... clear V

				clc

				rts

.7				dex
				dex
				bpl .1

				sec

.9				rts
*--------------------------------------
PFT.Y2MLIPath	ldx K.MLI.P1

.1				lda K.PathBuf,y
				beq .8

				iny

				sta K.MLI.P1+1,x
				inx
				cpx #MLI.MAXPATH+1
				bcc .1

				lda #MLI.E.INVPATH
*				sec
				rts

.8				stx K.MLI.P1

				clc
				rts
*--------------------------------------
