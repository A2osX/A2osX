PR#3
PREFIX /A2OSX.BUILD
LOMEM $A00
INC 1
AUTO 6
*/--------------------------------------
* # FAdd,FSub,FMult,FDiv
* Return X+Y,X-Y,X*Y,X/Y 
* ## In:
*  PUSHW = X (*Float)
*  PUSHW = Y (*Float)
*  Y,A = Dest (*Float)
*\--------------------------------------
K.FAdd			ldx #ROM.FADD
				.HS 2C					BIT ABS
K.FSub			ldx #ROM.FSUB
				.HS 2C					BIT ABS
K.FMult			ldx #ROM.FMULT
				.HS 2C					BIT ABS
K.FDiv			ldx #ROM.FDIV

				phx
				jsr MATH.PullFAC
				jsr MATH.PullARG
				>PULLW FORPNT
				plx
				jsr A2osX.ROMCALL
				>ROMCALL GETFAC
				clc
				rts
*/--------------------------------------
* # Log,Cos,Sin,Tan,ATan
* Return  
* ## In:
*  PUSHW = X (*Float)
*  Y,A = Dest (*Float)
*\--------------------------------------
K.Cos			jsr MATH.PullFAC
*				>ROMCALL COS	
*--------------------------------------
MATH.PullFAC	>PULLA
				tay
				>PULLA
				>ROMCALL SETFAC.AY
				rts
*--------------------------------------
MATH.PullARG	>PULLA
				tay
				>PULLA
				>ROMCALL SETARG.AY
				rts
*--------------------------------------
*--------------------------------------
