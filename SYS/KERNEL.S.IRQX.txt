*--------------------------------------
* called directly by IRQ Vector $FFFE in AuxLC0
* Must keep:
*  A,X,Y
*   IO.SETREADAUX
*  if we are here, AuxLC0 ZP/Stack is selected
* Exit with RTI
* if A2osX Drivers cannot clear IRQ,
* must call regular System Handler
*--------------------------------------
IRQX			cld

				sty IRQX.Y+1
				stx	IRQX.X+1
				sta IRQX.A+1

				pla						Y = P register
				pha

				and #$10				BRK?
				tay						Y = NZ if BRK

				lda IO.RDREADAUX
				ora #$40				IO.RDALTZP=1
				sta IO.CLRREADAUX

				jmp IRQ.XLC0
*--------------------------------------
* CC = IRQ Cleared
*--------------------------------------
IRQX.Exit		tax						DONT disturb C
				bpl IRQX.Y

				sta IO.SETREADAUX

IRQX.Y			ldy #$ff				SELF MODIFIED
IRQX.X			ldx #$ff				SELF MODIFIED
IRQX.A			lda #$ff				SELF MODIFIED

				bcs IRQX.JMP			Not handled, pass to ROM

				rti

IRQX.JMP		jmp $FFFF				SELF MODIFIED
*--------------------------------------
