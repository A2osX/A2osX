NEW
  AUTO 3,1
*--------------------------------------
* NULL DEVICE
*  - OPEN,WRITE,CLOSE always succeed.
*  - READ returns EOF.
*--------------------------------------
NULL.DRV	 	jmp (.1,x)

.1				.DA .10					IOCTL.STATUS
				.DA A2osX.BADCALL		IOCTL.READBLOCK
				.DA A2osX.BADCALL       IOCTL.WRITEBLOCK
				.DA A2osX.BADCALL       IOCTL.FORMAT
				.DA A2osX.BADCALL		IOCTL.CONTROL
				.DA A2osX.BADCALL		IOCTL.INIT
				.DA .80					IOCTL.OPEN
				.DA .80					IOCTL.CLOSE
				.DA .20					IOCTL.READ
				.DA .30					IOCTL.WRITE
*--------------------------------------
.10				>LDYAI NULL.DIB
				jmp DRV.STATUS.YA
*--------------------------------------
.20				lda #1
				sta K.S.IOCTL+S.IOCTL.BYTECNT
				stz K.S.IOCTL+S.IOCTL.BYTECNT+1

				jsr DRV.GetpBuf

				bit A2osX.IOOpt
				bpl .2

				sta IO.SETWRITEAUX

.2				lda #C.EOF
				sta (ZPpBuf)

				sta IO.CLRWRITEAUX

				clc
				rts
*--------------------------------------
.30				>LDYA K.S.IOCTL+S.IOCTL.BYTECNT

*				clc
*				rts
*--------------------------------------
.80				clc
				rts
*--------------------------------------
NULL.DIB		.DA #S.DIB.S.WRITE+S.DIB.S.READ+S.DIB.S.OPENED
				.DA #0,#0,#0
				.PS "NULL Device"
				.BS 5
				.DA #S.DIB.T.CHAR
				.DA #0
				.DA K.VER
*--------------------------------------
MAN
SAVE usr/src/sys/kernel.s.null
LOAD usr/src/sys/kernel.s
ASM
