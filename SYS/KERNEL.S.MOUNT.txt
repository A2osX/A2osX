* # GetStkObj
* ## C
* `int *ptr getstkobj (shoirt int hStkObj);`
* ## ASM
* `lda hStkObj`
* `>SYSCALL GetStkObj`
* ## RETURN VALUE
*  CC : success
*   X = hMem
*   Y,A = ptr
*  CS : error
*   A = EC
*\--------------------------------------
K.GetStkObj		tax						hMem for CopyToMain
				
				jsr MEM.GetMemByID
				
				ldy #S.MEM.LEN
				lda (ZPMemMgrSPtr),y
				pha
				iny
				lda (ZPMemMgrSPtr),y
				ply						YA = Len

				jmp MEM.CopyToMain
