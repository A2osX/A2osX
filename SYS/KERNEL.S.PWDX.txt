PWDX.GetPWUID	>LDYA PWDX.pUsrDB
				>STYA ZPPtr2

				sta IO.CLRWRITEAUX

				ldy #PWD.USR.L-1
				
.1				lda (ZPPtr2),y
				sta K.S.STAT,y
				dey
				bpl .1

*				sta IO.SETWRITEAUX

				clc

				rts

.98				lda #E.IPWDDB
				.HS 2C					BIT ABS
.99				lda #E.IUSR
				sec
				rts
