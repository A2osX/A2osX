M32.Add			ldx #0					ARG-ACC->ACC
				ldy #4
		
				clc
				
.1				lda M32.ARG,x
				adc	M32.ACC,x
				sta M32.ACC,x
				inx
				dey
				bne .1
			
				rts						if CS, Overflow
*---------------------------------------
M32.Sub			ldx #0					ARG-ACC->ACC
				ldy #4

				sec

.1				lda M32.ARG,x
				sbc	M32.ACC,x
				sta M32.ACC,x
				inx
				dey
				bne .1
				
				bcs .8					if CC, Overflow	
				
				sec
				rts
				
.8				clc
				rts
