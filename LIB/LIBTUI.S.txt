LIB.LOAD		lda LibCnt
				bne .8

				>LDYAI 512
				>SYSCALL2 GetMem
				bcs .9

				>STYA LineBuf
				stx hLineBuf



.8				inc LibCnt
				clc
.9				rts
*--------------------------------------
LIB.UNLOAD		dec LibCnt
				bne .8

				lda hLineBuf
				>SYSCALL2 FreeMem

.8				clc
				rts
*--------------------------------------
LIB.GetCTX		ldy #S.OBJ.hParent
				lda (ZPObjPtr),y
				
.1				>SYSCALL2 GetMemPtr
				>STYA ZPCtxPtr
				
				lda (ZPCtxPtr)
*				cmp #S.OBJ.T.SCRN
				beq .8
				
				ldy #S.OBJ.hParent
				lda (ZPCtxPtr),y
				bra .1
				
.8				rts
*--------------------------------------
LIB.GetOBJ		>SYSCALL2 GetMemPtr
				>STYA ZPObjPtr
				rts
