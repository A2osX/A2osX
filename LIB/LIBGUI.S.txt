LIB.LOAD		lda LibCount
				bne .8

				jsr FON.Init
				bcs .9

				jsr MOU.Init
				bcs .9

				jsr WND.Init
				bcs .9
				
				jsr WND.Setup
				
				lda hDevMouse
				beq .8
				
				jsr MOU.Setup
				
.8				inc LibCount

				clc
.9				rts
*--------------------------------------
LIB.UNLOAD		dec LibCount
				bne .8
				
				jsr MOU.Quit
				
				
				
.8				clc
				rts
