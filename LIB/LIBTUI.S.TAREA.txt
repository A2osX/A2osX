* TAREA.New
* (pStack)+6 pParent
* (pStack)+4 X1,Y1
* (pStack)+2 W,H
* (pStack)+0 pBuf
*--------------------------------------
TAREA.New		lda #S.OBJ.T.TAREA
				jsr OBJ.Create
				bcs .99

				>GETSTKW 6
				jsr OBJ.SetParent

				>GETSTKW 4
				jsr OBJ.SetX1Y1

				>GETSTKW 2
				jsr OBJ.SetWH

				>GETSTKW
				ldy #S.OBJ.pBuf
				sta (ZPpObj),y
				txa
				iny
				sta (ZPpObj),y

				ldy #S.OBJ.S
				lda #S.OBJ.S.bDisabled
				sta (ZPpObj),y

				jsr CONT.Insert
				bcs .99

				>LDYA ZPpObj

*				clc

.99				rts
*--------------------------------------
TAREA.Draw		lda (pStack)
				tax

				jsr OBJ.GetLineXAtY
				bcs .5

				ldy #S.OBJ.W
				lda (ZPpObj),y
				tax

				jsr LB.Ptr1OutEscMaxX
				txa
				beq .8

				bne .6

.5				ldy #S.OBJ.W
				lda (ZPpObj),y
				tax

.6				lda #C.SPACE

.7				jsr LB.COut
				dex
				bne .7

.8				clc
				rts
