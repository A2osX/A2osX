NEW
  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF bin/bf
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/a2osx.api.i
				.INB inc/mli.i
				.INB inc/mli.e.i
*--------------------------------------
				.MA PULLA
				lda (pStack)
				inc pStack
				.EM
*--------------------------------------
				.MA PULLW
				>PULLA
				sta ]1
				>PULLA
				sta	]1+1
				.EM
*--------------------------------------
BF.DATASIZE		.EQ 1024
*--------------------------------------
* Zero Page Segment, up to 32 bytes
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
pCodePtr		.BS 2
pDataPtr		.BS 2

ZPPtr1			.BS 2
ZPPtr2			.BS 2
ZPPtr3			.BS 2

ArgIndex		.BS 1

pCodeBuf		.BS 2
pDataBuf		.BS 2

bDebug			.BS 1
bTrace			.BS 1
StackCnt		.BS 1
Cnt				.BS 1
ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #0					#S.PS.SIG
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
* Relocation Table
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG
				.DA	CS.QUIT
*--------------------------------------
L.MSG.GREETINGS	.DA MSG.GREETINGS
L.MSG.USAGE		.DA MSG.USAGE
L.MSG.ECHOCRLF	.DA MSG.ECHOCRLF
L.MSG.DEBUG		.DA MSG.DEBUG
L.MSG.ERR		.DA MSG.ERR
J.BF.KW			.DA BF.KW.INCPTR
				.DA BF.KW.DECPTR
				.DA BF.KW.INCDATA
				.DA BF.KW.DECDATA
				.DA BF.KW.PUTCHAR
				.DA BF.KW.GETCHAR
				.DA BF.KW.WHILE
				.DA BF.KW.END
				.DA 0
*--------------------------------------
* Called once at process creation
* Put code for loading LIB here
*--------------------------------------
CS.INIT			clc
CS.INIT.RTS		rts
*--------------------------------------
* Called until exit with CS
* if RUN exits with CC, RN entered again
*--------------------------------------
CS.RUN			>SS
				>PUSHW L.MSG.GREETINGS
				>PUSHW A2osX.KVER
				>PUSHBI 2
				>LIBC PrintF
				>SR
				bcs CS.INIT.RTS

				jsr CS.RUN.ARGS
				bcs CS.INIT.RTS

				jsr BF.Init
				bcs CS.INIT.RTS

.1				>SLEEP

				bit bDebug
				bpl .2

				jsr PrintDebugMsg

.2				bit bTrace
				bpl .3

				jsr PrintTraceMsg

.3				jsr BF.Run
				bcc .1

				cmp #MLI.E.EOF
				beq .8

				pha
				jsr PrintErrMsg
				pla
				sec
				rts

.8				lda #0					Exit Code = Success
				sec
				rts
*--------------------------------------
CS.RUN.ARGS		inc ArgIndex

				lda ArgIndex
				>KAPI ArgV
				bcs .7

				>STYA ZPPtr1
				lda (ZPPtr1)
				cmp #'-'
				bne .4

				ldy #1
				lda (ZPPtr1),y

				ldx #OptionVars-OptionList-1

.1				cmp OptionList,x
				beq .2

				dex
				bpl .1

				bra .90

.2				ldy OptionVars,x
				lda #$ff
				sta $0,y
				bra CS.RUN.ARGS

.4				lda pCodeBuf+1
				bne .90

				>LDYA ZPPtr1
				jsr CS.RUN.LoadFile
				bcs .9

				bra CS.RUN.ARGS

.7				lda pCodeBuf+1
				beq .90

				clc
.9				rts

.90				>LDYA L.MSG.USAGE
				>LIBC PutS

				lda #E.SYN
				sec						QUIT Process
				rts
*--------------------------------------
CS.RUN.LoadFile	>LIBC LoadTxtFile
				bcs .9

				>STYA pCodeBuf
				>STYA pCode

.9				rts
*--------------------------------------
BF.Init			>LDYAI BF.DATASIZE
				>LIBC Malloc
				bcs .9

				>STYA pDataPtr
				>STYA pDataBuf

				>STYA ZPPtr1

				ldx /BF.DATASIZE
				lda #0
				tay

.1				sta (ZPPtr1),y
				iny
				bne .1

				inc ZPPtr1+1
				dex
				bne .1

.9				rts
*--------------------------------------
BF.Run			jsr BF.GetCodeByte
				beq .8

				ldx #BF.KW.Cnt-1

.2				cmp BF.KW,x
				bne .3

				txa
				asl
				tax
				jmp (J.BF.KW,x)

.3				dex
				bpl .2

				lda #E.CSYN
				sec
				rts

.8				lda #MLI.E.EOF

				sec
				rts
*--------------------------------------
BF.KW.INCPTR	inc pDataPtr
				bne .8

				inc pDataPtr+1

.8				clc
				jmp BF.NextCodeByte
*--------------------------------------
BF.KW.DECPTR	lda pDataPtr
				bne .8

				dec pDataPtr+1

.8				dec pDataPtr

				clc
				jmp BF.NextCodeByte
*--------------------------------------
BF.KW.INCDATA	lda (pDataPtr)
				inc
				sta (pDataPtr)
				clc
				jmp BF.NextCodeByte
*--------------------------------------
BF.KW.DECDATA	lda (pDataPtr)
				dec
				sta (pDataPtr)
				clc
				jmp BF.NextCodeByte
*--------------------------------------
BF.KW.PUTCHAR	lda (pDataPtr)
				>LIBC PutChar
				bcs .9

				jmp BF.NextCodeByte

.9				rts
*--------------------------------------
BF.KW.GETCHAR	>LIBC GetChar
				bcs .9

				sta (pDataPtr)

				jmp BF.NextCodeByte

.9				rts
*--------------------------------------
BF.KW.WHILE		lda (pDataPtr)
				bne .8

				lda #1
				sta Cnt

.1				jsr BF.NextCodeByte

				jsr BF.GetCodeByte
				beq .9

				cmp #'['
				bne .2

				inc Cnt
				bra .1

.2				cmp #']'
				bne .1

				dec Cnt
				bne .1

				clc
				jmp BF.NextCodeByte

.8				>PUSHW pCodePtr
				inc StackCnt
				clc
				jmp BF.NextCodeByte

.9				lda #E.ESYN
				sec
				rts
*--------------------------------------
BF.KW.END		lda StackCnt
				beq .9

				>PULLW pCodePtr
				dec StackCnt
				clc
				rts

.9				lda #E.STACKERROR
				sec
				rts
*--------------------------------------
BF.GetCodeByte	lda (pCodePtr)
				beq .8

				cmp #C.SPACE
				beq .7

				cmp #C.CR
				beq .7

				cmp #C.LF
				beq .7

				cmp #'#'
				bne .8

.2				jsr BF.NextCodeByte

				lda (pCodePtr)
				beq .8

				cmp #C.CR
				bne .2

.7				jsr BF.NextCodeByte

				bra BF.GetCodeByte

.8				rts
*--------------------------------------
BF.NextCodeByte	inc pCodePtr
				bne .8

				inc pCodePtr+1

.8				rts
*--------------------------------------
CS.SIG			sec
				rts
*--------------------------------------
CS.QUIT			>LDYA pCodeBuf
				beq .1

				>LIBC Free

.1				>LDYA pDataBuf
				beq .8

				>LIBC Free

.8				rts
*--------------------------------------
PrintDebugMsg	>LDYA pStack
				>STYA ZPPtr2

				>SS
				>PUSHW L.MSG.DEBUG

				>PUSHW pCodePtr
				>PUSHW pDataPtr
				>PUSHB (pDataPtr)
				>PUSHW ZPPtr2

				>PUSHBI 7
				>LIBC PrintF
				>SR

				rts
*--------------------------------------
PrintErrMsg		bit bTrace
				bmi .5

				jsr PrintTraceMsg

.5				lda ZPPtr3
				cmp pCodePtr
				bne .6

				lda ZPPtr3+1
				cmp pCodePtr+1
				beq .8

.6				>SS
				>PUSHWI '-'
				jsr STDERR
				>LIBC FPutC
				>SR

				inc ZPPtr3
				bne .5

				inc ZPPtr3+1
				bra .5

.8				>SS
				jsr STDERR
				>PUSHW L.MSG.ERR
				>PUSHW ZPPtr2			Line counter
				>PUSHBI 2
				>LIBC FPrintF
				>SR
				rts
*--------------------------------------
PrintTraceMsg	>LDYA pCodeBuf
				>STYA ZPPtr1

				stz ZPPtr2
				stz ZPPtr2+1			Line counter

.1				inc ZPPtr2
				bne .2

				inc ZPPtr2+1

.2				>LDYA ZPPtr1
				>STYA ZPPtr3			save line start

.20				lda (ZPPtr1)
				beq .4					EoF

				inc ZPPtr1
				bne .3

				inc ZPPtr1+1
.3				cmp #C.CR
				bne .20					Scan until EoL

				ldx ZPPtr1
				cpx pCodePtr
				lda ZPPtr1+1
				sbc pCodePtr+1
				bcc .1					not this line....

.4				>SS
				>PUSHWI '>'
				jsr STDERR
				>LIBC FPutC
				>SR
				ldy #$ff

.7				iny
				lda (ZPPtr3),y
				beq .8

				cmp #C.CR
				beq .8

				phy
				>SS
				pha
				>PUSHBI 0
				pla
				>PUSHA
				jsr STDERR
				>LIBC FPutC
				>SR
				ply
				bra .7

.8 				>SS
				jsr STDERR
				>PUSHW L.MSG.ECHOCRLF
				>PUSHBI 0
				>LIBC FPrintF
				>SR
				rts
*--------------------------------------
STDERR			ldy #S.PS.pStdErr+1
				jsr A2osX.GetPSy
				>PUSHA
				dey
				jsr A2osX.GetPSy
				>PUSHA
				rts
*--------------------------------------
CS.END			.EQ *
*--------------------------------------
* Initialized DATA
*--------------------------------------
MSG.GREETINGS	.CZ "\r\nBrainFuck-Shell %d.%02d\r\n\r\n"
MSG.USAGE		.CS "Usage : BF <option> file\r\n"
				.CS "   -D : Debug Mode\r\n"
				.CS "   -T : Trace On"
MSG.ECHOCRLF	.CZ "\r\n"
MSG.DEBUG		.CZ "pCode=%H, pData=%H(%h), pStack=%H\r\n"
MSG.ERR			.CZ "-^\r\nLine #%D:"
*--------------------------------------
OptionList		.AS "DdTt"
OptionVars		.DA #bDebug,#bDebug,#bTrace,#bTrace
*--------------------------------------
BF.KW			.AS "><+-.,[]"
BF.KW.Cnt		.EQ *-BF.KW
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0

DS.END			.ED
*--------------------------------------
MAN
SAVE usr/src/bin/bf.s
ASM
