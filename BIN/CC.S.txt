CS.RUN			>PUSHW L.MSG.GREETINGS

				>SYSCALL PrintF
				bcs CS.INIT.RTS

				txa

				bcs CS.INIT.RTS

				jsr CC.CORE.Init
				bcs CS.INIT.RTS

				jsr CODE.Init

				jsr CC.FIO.FOpen1st
				bcs CS.INIT.RTS

				cmp #MLI.E.EOF
				bne .9

				bcs .99

				jsr CC.Link

				bcs .99

				dec bPass2

				jsr CC.FIO.FCreate
				bcs .99

				jsr CODE.Init
				bcs .99

				jsr CC.FIO.FOpen1st
				bcs .99

CS.RUN.ARGS		inc ArgIndex

				lda ArgIndex
				>SYSCALL ArgV
				bcs .8

				>STYA ZPPtr1
				lda (ZPPtr1)
				cmp #'-'
				bne .4

				ldy #1
				lda (ZPPtr1),y

				ldx #OptionVars-OptionList-1

.1				cmp OptionList,x
				beq .2

				dex
				bpl .1

				bra .90

.2				ldy OptionVars,x
				lda #$ff
				sta (pData),y
				bra CS.RUN.ARGS

.4				>LDA.G CC.InFileArg
				bne .5

				lda ArgIndex
				>STA.G CC.InFileArg
				bra CS.RUN.ARGS

.5				>LDA.G CC.OutFileArg
				bne .90

				lda ArgIndex
				>STA.G CC.OutFileArg
				bra CS.RUN.ARGS

.8				>LDA.G CC.InFileArg
				beq .90

				>LDA.G CC.OutFileArg
				beq .90

				clc
.9				rts

.90				>PUSHW L.MSG.USAGE
				>PUSHBI 0
				>SYSCALL PrintF

				lda #E.SYN
				sec
				rts

				rts

				>PUSHW ZPCCCode
				>PUSHW ZPCCConst
				>PUSHW ZPCCData

				lda pStack
				clc
				adc #8
				pha
				lda pStack+1
				adc #0
				>PUSHA
				pla
				>PUSHA

				>PUSHB LocalPtr

				>PUSHW ZPCCStack

				>PUSHB CStackPtr
				tay
				beq .1

				lda (ZPCCStack),y
				pha
				iny
				lda (ZPCCStack),y
				ply

.1				>PUSHYA

				lda ScopeIdx
				sec
				sbc #CC.hScopes
				>PUSHA

				>PUSHBI 15
				>SYSCALL PrintF

				rts

				>STYA ZPLineBufPtr

				bmi .1

				txa
