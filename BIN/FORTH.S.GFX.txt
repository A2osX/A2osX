GFX.Open		>SS
				>PUSHW L.DEV.GFX
				>PUSHWI 0
				>LIBC Open
				>SR
				bcs .9

				>STYA.G hDevGFX

				bra GFX.TEXT

.9				rts
*--------------------------------------
GFX.Close		>LDYA.G hDevGFX
				beq .9

				>LIBC Close

.9				rts
*--------------------------------------
GFX.TEXT		ldy #S.PS.pStdIn
				jsr A2osX.GetPSy
				pha
				iny
				jsr A2osX.GetPSy
				ply

				>LIBC FileNo
				bra GFX.IOCTL

GFX.TEXT.RTS	clc
				rts
*--------------------------------------
GFX.GR			>LDYA.G hDevGFX
				beq GFX.TEXT.RTS

GFX.IOCTL		>SS
				>PUSHYA
				>PUSHBI IOCTL.CONTROL
				>PUSHWZ
				>LIBC IOCTL
				>SR
				rts
*--------------------------------------
* (Y X C)
*--------------------------------------
GFX.PLOT		ldx #S.CB.CMD.SETPIXEL
				jsr GFX.WriteX

				lda pStack
				clc
				adc #8
				sta pStack
				clc
				rts
*--------------------------------------
* (Y2 X2 Y1 X1 C)
*--------------------------------------
GFX.RECT		ldx #S.CB.CMD.FILLRECT

				jsr GFX.WriteX

				lda pStack
				clc
				adc #12
				sta pStack
				clc
				rts
*--------------------------------------
GFX.WriteX		ldy #1
				lda (pStack)
				sta (pStack),y

				lda #S.CB.M.C16
				sta (pStack)
				txa
				tay

				>SS
				>PUSHBI S.CB.OP.SET
				tya
				>PUSHA

				>PUSHW.G hDevGFX
				>PUSHBI IOCTL.WRITE
				lda pStack
				clc
				adc #2
				tay
				lda pStack+1
				adc #0
				>PUSHYA
				>LIBC IOCTL
				>SR
			clc							DHGR.DRV bug ?
				rts
