MAC.Init			>SYSCALL SListNew
					bcs .9

					>STA.G MAC.hMacros
							
					>LDYAI 256
					>SYSCALL GetMem.YA
					bcs .9
					
					>STYA ZPMacroBuf
					txa
					>STA.G MAC.hBuf
					
.9					rts					
*---------------------------------------
MAC.Quit			>LDA.G MAC.hBuf
					beq .1

					>SYSCALL FreeMem.A
					
.1					>LDA.G MAC.hMacros
					beq .8
					>SYSCALL SListFree.A

.8					clc
					rts			
*---------------------------------------
MAC.Clear			lda #0
					tay
					
.1					sta (ZPMacroBuf),y
					iny
					bne .1
					
					rts
