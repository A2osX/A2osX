NEW
  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF bin/adt
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/a2osx.api.i
				.INB inc/mli.e.i
				.INB inc/eth.i
				.INB inc/ip.i
				.INB inc/net.skt.i
				.INB inc/net.db.i
*--------------------------------------
RESP.BUFSIZE	.EQ 1024
TIMEOUT.MAX		.EQ 20					20 sec.
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START

ZPpRespBuf		.BS 2
ZPRespLen		.BS 2

ZPpPayloadBuf	.BS 2
ZPPayloadLen	.BS 2

ZPCount			.BS 2

hSocket			.BS 2

iArg			.BS 1
iHost			.BS 1
iPort			.BS 1
TimeOut			.BS 1

ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #S.PS.SIG.T1SEC
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG
				.DA	CS.QUIT
*--------------------------------------
J.CheckArgs		.DA CS.RUN.CheckArgs.U
				.DA CS.RUN.CheckArgs.U
				.DA CS.RUN.CheckArgs.F
				.DA CS.RUN.CheckArgs.F

L.MSG.USAGE		.DA MSG.USAGE

L.MSG.UNKNOWN	.DA MSG.UNKNOWN
L.MSG.HOSTOK	.DA MSG.HOSTOK
L.MSG.SKTKO		.DA MSG.SKTKO
L.MSG.SKTOK		.DA MSG.SKTOK
L.MSG.SKTERR	.DA MSG.SKTERR
L.MSG.IOERR		.DA MSG.IOERR
L.ADT.DIR		.DA ADT.DIR

L.SA.LOCAL		.DA SA.LOCAL

				.DA 0
*--------------------------------------
CS.INIT			clc
				rts
*--------------------------------------
CS.RUN			jsr CS.RUN.CheckArgs
				bcs .9

				jsr CS.RUN.GetHost
				bcs .9

				jsr CS.RUN.GetPort
				bcs .9

				jsr CS.RUN.OpenSocket
				bcs .9

				>LDYAI RESP.BUFSIZE
				>LIBC Malloc
				bcs .9

				>STYA ZPpRespBuf
				>STYA.G pRespBuf

				jsr CS.RUN.GETDIR
				bcs .9

.8				lda #0
				sec
.9				rts
*--------------------------------------
CS.RUN.CheckArgs
				inc iArg
				lda iArg
				>KAPI ArgV
				bcs .8

				>STYA R1
				lda (R1)
				cmp #'-'
				bne .4

				ldy #2
				lda (R1),y
				bne .9

				dey
				lda (R1),y

				ldx #OptionCnt-1

.1				cmp OptionList,x
				beq .2

				dex
				bpl .1

.9				>LDYA L.MSG.USAGE
				>LIBC PutS

				lda #E.SYN
				sec
				rts

.2 				txa
				asl
				tax
				jsr .3
				bcs .9

				bra CS.RUN.CheckArgs

.3				jmp (J.CheckArgs,x)
*--------------------------------------
.4				lda iHost
				bne .5

				lda iArg
				sta iHost
				bra CS.RUN.CheckArgs

.5				lda iPort
				bne .9

				lda iArg
				sta iPort
				bra CS.RUN.CheckArgs

.8				lda iHost
				beq .9

				clc
				rts
*--------------------------------------
CS.RUN.CheckArgs.U
CS.RUN.CheckArgs.F

				sec
				rts
*--------------------------------------
CS.RUN.GetHost	lda iHost
				>KAPI ArgV
				>STYA R1

				lda #TIMEOUT.MAX
				sta TimeOut

.2				>SLEEP

				>SS
				>PUSHW R1
				>PUSHWZ
				>PUSHWZ
				>PUSHEA.G pAI
				>LIBC GetAddrInfo
				>SR
				bcc .4

				cmp #GAI.E.AGAIN
				bne .3

				lda TimeOut
				bne .2

.3				>SS
				>PUSHW L.MSG.UNKNOWN
				>PUSHW R1
				>PUSHBI 2
				>LIBC PrintF
				>SR

.9				lda #E.SYN
				sec
				rts

.4				>LDYA.G pAI
				>STYA R2

				ldy #S.ADDRINFO.pSOCKADDR
				lda (R2),y
				sta R3
				iny
				lda (R2),y
				sta R3+1

				>LEA.G SA.REMOTE
				>STYA R4

				ldy #S.SA.IN-1

.5				lda (R3),y
				sta (R4),y
				dey
				bpl .5

*				clc

				rts
*--------------------------------------
CS.RUN.GetPort	lda iPort
				bne .1

				>LDYAI 6502
				bra .8

.1				>KAPI ArgV
				>LIBC AToI

.8				>STYA.G SA.REMOTE+S.SA.IN.PORT
				clc
				rts
*--------------------------------------
CS.RUN.OpenSocket
				lda #TIMEOUT.MAX
				sta TimeOut

				>SS
				>PUSHWI AF_INET
				>PUSHWI SOCK_DGRAM
				>PUSHWZ
				>LIBC Socket
				>SR
				bcs .9

				>STYA hSocket

				>SS
				>PUSHW hSocket
				>PUSHW L.SA.LOCAL
				>PUSHWI S.SA.IN
				>LIBC Bind
				>SR
				bcs .9

.3				>SLEEP

				>SS
				>PUSHW hSocket
				>PUSHEA.G SA.REMOTE
				>PUSHWI S.SA.IN
				>LIBC Connect
				>SR
				bcc .4

				cmp #SKT.E.PENDING
				bne .9

				ldx TimeOut
				bne .3

				bra .9

.4				clc
				rts

.9				pha
				tay
				>SS
				>PUSHW L.MSG.SKTERR
				tya
				>PUSHA
				>PUSHBI 1
				>LIBC PrintF
				>SR
				pla
				sec
.8				rts
*--------------------------------------
CS.RUN.GETDIR	>SS
				>PUSHW hSocket
				>PUSHW L.ADT.DIR
				>PUSHWI ADT.DIR.LEN
				>PUSHWZ					flags
				>LIBC Send
				>SR
				bcc .2

				cmp #SKT.E.PENDING
				beq .2

				lda #MLI.E.IO
				sec
				rts

.2				lda #TIMEOUT.MAX
				sta TimeOut

.3				>SLEEP

				>SS
				>PUSHW hSocket
				>LDYA.G pRespBuf
				>STYA ZPpRespBuf
				>PUSHYA
				>PUSHWI RESP.BUFSIZE
				>PUSHWZ					flags
				>LIBC Recv
				>SR
				bcc CS.RUN.GETDIR.R

				ldx TimeOut
				bne .3

*				sec

.9				rts
*--------------------------------------
CS.RUN.GETDIR.R	>STYA ZPRespLen

				ldy #S.UDP+1
				lda (pRespBuf),y
				cmp #$C1
				bne .9

				iny
				lda (pRespBuf),y
				sta ZPPayloadLen
				iny
				lda (pRespBuf),y
				sta ZPPayloadLen+1
				iny
				lda (pRespBuf),y
				cmp #$D3
				bne .9

				jsr CS.RUN.UNRLE
				bcs .99

				>SS
				ldy #S.PS.pStdOut+1
				jsr A2osX.GetPSy
				>PUSHA
				dey
				jsr A2osX.GetPSy
				>PUSHA
				>PUSHW pPayloadBuf
				>PUSHW ZPPayloadLen
				>LIBC fwrite
				>SR

				rts

.9				lda #MLI.E.IO

				sec

.99				rts
*--------------------------------------
CS.RUN.UNRLE	>LDYA ZPPayloadLen
				>LIBC Malloc
				bcs .9

				>STYA ZPpPayloadBuf
				>STYA.G pPayloadBuf

				lda ZPPayloadLen
				eor #$ff
				sta ZPCount

				lda ZPPayloadLen+1
				eor #$ff
				sta ZPCount+1

				ldx #0
				ldy #S.UDP+8

.1				inc ZPCount
				bne .2

				inc ZPCount+1
				beq .8

.2				txa
				clc
				adc (pRespBuf),y
				tax
				jsr .80

				iny
				bne .3

				inc pRespBuf+1
.3				bra .1

.8				clc

.9				rts

.80				sta (ZPpPayloadBuf)
				inc ZPpPayloadBuf
				bne .81

				inc ZPpPayloadBuf+1

.81				rts
*--------------------------------------
CS.SIG			lda TimeOut
				beq .9

				dec TimeOut

.9				rts
*--------------------------------------
CS.QUIT			ldy hSocket
				beq .1

				lda hSocket+1

				>LIBC Shutdown

.1				>STYA.G pRespBuf
				cmp #0
				beq .2

				>LIBC Free

.2

.8				clc
				rts
*--------------------------------------
CS.END			.EQ *
*--------------------------------------
OptionList		.AS "UuFf"
OptionCnt		.EQ *-OptionList
*--------------------------------------
MSG.USAGE		.CS "Usage : ADT <ip|host> [port]\r\n"
				.CS "        -U Url\r\n"
				.CZ "        -F UrlFile\r\n"
*--------------------------------------
MSG.UNKNOWN		.CZ "%s: Unknown host\r\n"
MSG.HOSTOK		.CZ "Connecting to %d.%d.%d.%d:%D (%s)..."
MSG.SKTKO		.CZ "Failed to Open Socket."
MSG.SKTOK		.CZ "Connected."
MSG.SKTERR		.CZ "Socket Error : $%h\r\n"
MSG.IOERR		.CZ "I/O Error : $%h\r\n"
*--------------------------------------
SA.LOCAL		.DA AF_INET
				.DA #0,#0,#0,#0			any address
				.DA 0					Dyn port
*--------------------------------------
ADT.DIR			.HS C10100C4040101
ADT.DIR.LEN		.EQ *-ADT.DIR
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0

pAI				.BS 2
SA.REMOTE		.BS S.SA.IN

pRespBuf		.BS 2
pPayloadBuf		.BS 2

DS.END			.ED
*--------------------------------------
