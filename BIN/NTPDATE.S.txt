NEW
  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF bin/ntpdate
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/a2osx.api.i
				.INB inc/mli.i
				.INB inc/eth.i
				.INB inc/ip.i
				.INB inc/net.skt.i
				.INB inc/net.db.i
*--------------------------------------
TIMEOUT.MAX		.EQ 25					25 sec.
BUFSIZE			.EQ 256
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
ZPPtr			.BS 2
hSocket			.BS 2
pBuf			.BS 2
BufLen			.BS 2

TimeOut			.BS 1
iArg			.BS 1

iHost			.BS 1
bSet			.BS 1
bDaemon			.BS 1

Time1900		.BS 4
Adjust			.BS 4

STime			.BS S.TIME

ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #S.PS.SIG.T1SEC
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG
				.DA	CS.QUIT
*--------------------------------------
J.CheckArgs		.DA CheckArgs.A
				.DA CheckArgs.S
				.DA CheckArgs.D
L.MSG.USAGE		.DA MSG.USAGE
L.MSG.UNKNOWN	.DA MSG.UNKNOWN
L.MSG.SOCKET	.DA MSG.SOCKET
L.MSG.TIMEOUT	.DA MSG.TIMEOUT
L.MSG.TIME		.DA MSG.TIME
L.SA.LOCAL		.DA SA.LOCAL
L.NTP.Request	.DA NTP.Request
				.DA 0
*--------------------------------------
CS.INIT			clc
				rts
*--------------------------------------
CS.RUN			jsr CheckArgs
				bcs .9

				jsr GetHost
				bcs .9

				jsr OpenSocket
				bcs .9

				>LDYAI BUFSIZE
				>LIBC Malloc
				bcs .9

				>STYA pBuf

.1				jsr SendNTPRequest
				bcs .2

				jsr DecodeNTPReply
				bcs .2

				bit bSet
				bpl .2

				jsr SetProDOSTime

.2				bit bDaemon
				bpl .7

				lda #TIMEOUT.MAX
				sta TimeOut

.3				>SLEEP

				lda TimeOut
				bne .3

				bra .1

.7				bcc .8

				rts

.8				lda #0
				sec
.9				rts
*--------------------------------------
CS.SIG			lda TimeOut
				beq .9

				dec TimeOut

.9				sec						do not discard TIMER event
				rts
*--------------------------------------
CS.QUIT			ldy hSocket
				beq .1

				>SS
				>PUSHW hSocket
				>PUSHWI 0				how
				>LIBC Shutdown
				>SR

.1				>LDYA pBuf
				beq .8

				>LIBC Free

.8				clc
				rts
*--------------------------------------
CheckArgs		inc iArg
				lda iArg
				>KAPI ArgV
				bcs .8

				>STYA ZPPtr
				lda (ZPPtr)
				cmp #'-'
				bne .4

				ldy #2
				lda (ZPPtr),y
				bne .9

				dey
				lda (ZPPtr),y

				ldx #OptionCnt-2

.2				cmp OptionList,x
				beq .3

				cmp OptionList+1,x
				beq .3

				dex
				dex
				bpl .2

				bra .9

.3				inc iArg

				jsr .5
				bcc CheckArgs
				bcs .9

.4				lda iHost
				bne .9

				lda iArg
				sta iHost
				bra CheckArgs

.5				jmp (J.CheckArgs,x)

.8				lda iHost
				beq .9

				clc
				rts

.9				>LDYA L.MSG.USAGE
				>LIBC PutS

				lda #E.SYN
				sec
				rts

CheckArgs.A		lda iArg
				>KAPI ArgV
				bcs .9

				>LIBC AToL

				>PUSHWZ
				>PUSHWI 3600
				>FPU uMUL
				>PULLL Adjust

				clc

.9				rts

CheckArgs.S		sec
				ror bSet
				rts

CheckArgs.D		sec
				ror bDaemon
				rts
*--------------------------------------
GetHost			lda #TIMEOUT.MAX
				sta TimeOut

				lda iHost
				>KAPI ArgV

				>STYA R1

.2				>SLEEP

				>SS
				>PUSHW R1
				>PUSHWZ
				>PUSHWZ
				>PUSHEA.G pAI
				>LIBC GetAddrInfo
				>SR
				bcc .4

				cmp #GAI.E.AGAIN
				bne .3

				lda TimeOut
				bne .2

.3				>SS
				>PUSHW L.MSG.UNKNOWN
				>PUSHW R1
				>PUSHBI 2
				>LIBC PrintF
				>SR

.9				lda #E.SYN
				sec
				rts

.4				>LDYA.G pAI
				>STYA R1

				ldy #S.ADDRINFO.pSOCKADDR
				lda (R1),y
				sta R2
				iny
				lda (R1),y
				sta R2+1

				>LEA.G SA.REMOTE
				>STYA R3

				ldy #S.SA.IN-1

.5				lda (R2),y
				sta (R3),y
				dey
				bpl .5

				>LDYAI UDP.PORT.NTP
				>STYA.G SA.REMOTE+S.SA.IN.PORT

*				clc

				rts
*--------------------------------------
OpenSocket		>SS
				>PUSHWI AF_INET
				>PUSHWI SOCK_DGRAM
				>PUSHWZ
				>LIBC Socket
				>SR
				bcs .9

				>STYA hSocket

				>SS
				>PUSHW hSocket
				>PUSHW L.SA.LOCAL
				>PUSHWI S.SA.IN
				>LIBC Bind
				>SR
				bcs .9

				>SLEEP

				>SS
				>PUSHW hSocket
				>PUSHEA.G SA.REMOTE
				>PUSHWI S.SA.IN
				>LIBC Connect
				>SR
				bcc .8

.9				pha
				>PUSHW L.MSG.SOCKET
				>PUSHBI 0
				>LIBC PrintF
				pla
				sec
.8				rts
*--------------------------------------
SendNTPRequest	lda #TIMEOUT.MAX
				sta TimeOut

				>SS
				>PUSHW hSocket
				>PUSHW L.NTP.Request
				>PUSHWI NTP.Request.End-NTP.Request
				>PUSHWZ					flags
				>LIBC Send
				>SR
				bcc .1

				cmp #SKT.E.PENDING
				beq .1

				sec
				rts

.1				>SLEEP

				>SS
				>PUSHW hSocket
				>PUSHW pBuf
				>PUSHWI BUFSIZE
				>PUSHWZ					flags
				>LIBC Recv
				>SR
				bcc .8

				ldx TimeOut
				bne .1

.9				>SS
				>PUSHW L.MSG.TIMEOUT
				lda iHost
				>KAPI ArgV
				>PUSHYA
				>PUSHBI 2
				>LIBC PrintF
				>SR
				sec
.8				rts
*--------------------------------------
DecodeNTPReply	>STYA BufLen

				lda #40

				clc
				adc pBuf
				sta ZPPtr
				lda /40
				adc pBuf+1
				sta ZPPtr+1

				ldy #3
				ldx #0

				clc

.1				lda	(ZPPtr),y
				adc Adjust,x
				sta Time1900,x
				inx
				dey
				bpl .1

				>SS
				>PUSHWI Time1900
				>PUSHWI STime
				>LIBC CTime2Time
				>SR

				lda STime+S.TIME.YEAR
				sec
				sbc #70
				sta STime+S.TIME.YEAR

				>SS
				>PUSHW L.MSG.TIME

				ldy #0

.2				>PUSHB STime,y
				iny
				cpy #7
				bne .2

				>PUSHBI 7
				>LIBC PrintF
				>SR

				rts
*--------------------------------------
SetProDOSTime	lda STime+S.TIME.MONTH
				asl
				asl
				asl
				asl
				asl
				sta MLI.DATE

				lda STime+S.TIME.YEAR
				rol
				sta MLI.DATE+1

				lda STime+S.TIME.DAY
				ora MLI.DATE
				sta MLI.DATE

				lda STime+S.TIME.HOUR
				sta MLI.TIME+1

				lda STime+S.TIME.MINUTE
				sta MLI.TIME

				clc
				rts
*--------------------------------------
CS.END			.EQ *
*--------------------------------------
OptionList		.AS "AaSsDd"
OptionCnt		.EQ *-OptionList
*--------------------------------------
MSG.USAGE		.CS "Usage : NTPDATE <ip|host>\r\n"
				.CS "   -A (-)nn : TZ offset\r\n"
				.CS "   -S : Set ProDOS date\r\n"
				.CZ "   -D : Run as Daemon\r\n"
MSG.UNKNOWN		.CZ "%s: Unknown host\r\n"
MSG.SOCKET		.CZ "Failed to open socket\r\n"
MSG.TIMEOUT		.CZ "No response from %s\r\n"
MSG.TIME		.CS "NTPDATE:%02d%02d/%02d/%02d %02d:%02d:%02d\r\n"
*--------------------------------------
SA.LOCAL		.DA AF_INET
				.DA #0,#0,#0,#0			any address
				.DA 0					Dyn port
*--------------------------------------
NTP.Request		.HS 1B
				.BS 47
NTP.Request.End	.EQ *
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0

pAI				.BS 2
SA.REMOTE		.BS S.SA.IN

DS.END			.ED
*--------------------------------------
MAN
SAVE usr/src/bin/ntpdate.s
ASM
