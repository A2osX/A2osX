					.LIST OFF

*---------------------------------------
* Addressing Mode reserved : #(),XY
* U-Operators :
* B-Operators :
* Num Format : 0-9 -> decimal
* 				$ = Hexa
*				% = binary
*               & = octal
*---------------------------------------
EXP.Get				jsr SRC.GetCharNB
					beq .9
					jsr SRC.IsDigit10		Decimal constant ?
					bne .
					
					
					
					
					
					
					cmp #'$'			Hex?
					bne .
					
					
					
					
					cmp #'''			literal?
					bne .
					
					
					
					
					jsr SRC.IsLetter		Symbol ?
					bne .
					
					cmp #'#'
.9					lda #ERR.MISSING.EXP
					sec
.99					rts
*---------------------------------------
					stz SRC.ACC
					stz SRC.ACC+1
					stz SRC.ACC+2
					stz SRC.ACC+3
*---------------------------------------					
EXP.Operators		.AS "^!|&<=>+-*/"
