CSH.BOP.EXEC	cpy #CSH.T.UINT
				bcs .3

				cpy #CSH.T.UCHAR
				beq .1

				cpy #CSH.T.SCHAR
				beq .2

				lda #E.TMISMATCH
				sec
				rts

.1				tax
				jmp (J.CSH.bBOPS,x)

.2				tax
				jmp (J.CSH.cBOPS,x)

.3				clc				
				adc CSH.Type2FPU-CSH.T.UINT,y
				tax
				jsr A2osX.FPUCALL
				
				clc
				rts

.8				inc pStack

.8				inc pStack
