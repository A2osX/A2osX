CS.RUN			>SS
				>PUSHW L.MSG.GREETINGS
				>PUSHW A2osX.KVER
				>PUSHBI 2
				>LIBC PrintF
				>SR
				bcs CS.INIT.RTS

				jsr CS.RUN.ARGS
				bcs CS.INIT.RTS

				jsr CSH.Init
				bcs CS.INIT.RTS

.1				>SLEEP

				>LDA.G bDebug
				bpl .2

				jsr PrintDebugMsg

.2				>LDA.G bTrace
				bpl .3

				>LDYA ZPInputBufPtr
				jsr PrintTraceMsg

.3				jsr CSH.Run
				bcc .1

				cmp #MLI.E.EOF
				beq .8

				pha
				jsr PrintErrMsg
				pla
				sec
				rts

.8				lda #0					Exit Code = Success
				sec
				rts
*--------------------------------------
CS.RUN.ARGS		lda #0
				>KAPI ArgV
				>LIBC PutS

				ldy #S.PS.pCmdLine
				jsr A2osX.GetPSy
				pha
				iny
				jsr A2osX.GetPSy
				ply
				>LIBC PutS

				lda #0
				>KAPI ArgV
				>STYA ZPPtr1

				>SS
				>PUSHYA
				ldy #S.PS.pCmdLine+1
				jsr A2osX.GetPSy
				>PUSHA
				dey
				jsr A2osX.GetPSy
				>PUSHA
				>LIBC StrCmp
				>SR
				bcs .4

.10				inc ArgIndex

				lda ArgIndex
				>KAPI ArgV
				bcs .7

				>STYA ZPPtr1
				lda (ZPPtr1)
				cmp #'-'
				bne .4

				ldy #1
				lda (ZPPtr1),y

				ldx #OptionVars-OptionList-1

.1				cmp OptionList,x
				beq .2

				dex
				bpl .1

				bra .90

.2				ldy OptionVars,x
				lda #$ff
				sta (pData),y
				bra .10

.4				>LDA.G CSH.pFileBuf+1
				bne .90

				>LDYA ZPPtr1

.5				>LIBC LoadTxtFile
				bcs .9

				>STYA ZPInputBufPtr
				>STYA.G CSH.pFileBuf
				bra .10

.7				>LDA.G CSH.pFileBuf+1
				beq .90

				clc
.9				rts

.90				>LDYA L.MSG.USAGE
				>LIBC PutS

				lda #E.SYN
				sec						QUIT Process
				rts

CS.QUIT			jsr CSH.Quit
				clc
				rts
*--------------------------------------
PrintErrMsg		>LDYA.G CSH.pDefineBuf
				cmp #0
				beq .10

				>LDYA.G CSH.SaveDefine
				>STYA ZPInputBufPtr

.10				>LDYA.G CSH.pIncludeBuf
				cmp #0
				beq .11

				bra .12

.11				>LDYA.G CSH.pFileBuf

.12				>STYA ZPPtr1

				stz ZPPtr2
				stz ZPPtr2+1			Line counter

.1				>INCW ZPPtr2

				>LDYA ZPPtr1
				>STYA ZPPtr3			save line start

.20				lda (ZPPtr1)
				beq .4					EoF

				>INCW ZPPtr1

				cmp #C.CR
				bne .20					Scan until EoL

				ldx ZPPtr1
				cpx ZPInputBufPtr
				lda ZPPtr1+1
				sbc ZPInputBufPtr+1
				bcc .1					not this line....

.4				>LDA.G bTrace
				bmi .5

				jsr PrintTraceMsg.3

.5				lda ZPPtr3
				cmp ZPInputBufPtr
				bne .6

				lda ZPPtr3+1
				cmp ZPInputBufPtr+1
				beq .8

.6				>SS
				>PUSHWI '-'
				jsr STDERR
				>LIBC FPutC
				>SR

				inc ZPPtr3
				bne .5

				inc ZPPtr3+1
				bra .5

.8				>SS
				jsr STDERR
				>PUSHW L.MSG.ERR
				>PUSHW ZPPtr2			Line counter
				>PUSHBI 2
				>LIBC FPrintF
				>SR

				rts

				>PUSHW L.MSG.DEBUG
				>PUSHW ZPPtr2
				>PUSHBI 2
				>LIBC PrintF
				>SR

				rts
*--------------------------------------

*			>DEBUG
PrintTraceMsg.3	>SS
				>PUSHWI '>'
				jsr STDERR
				>LIBC FPutC
				>SR

.1				iny
				lda (ZPPtr3),y
				beq .8

				cmp #C.CR
				beq .8

				phy

				tay
				lda #0

				>SS
				>PUSHYA
				jsr STDERR
				>LIBC FPutC
				>SR

				ply

				bra .1

.8 				>SS
				jsr STDERR
				>PUSHW L.MSG.ECHOCRLF
				>PUSHBI 0
				>LIBC FPrintF
				>SR

				rts
