RT.StrSet		sty STRID
				stx STRID+1

				ldy #1
				lda (pStack),y
				ora (pStack)
				beq .1

				>PUSHW.G pStrings

				ldy #2
				lda (pStack),y
				>PUSHA
				lda (pStack),y
				>PUSHA

				>PUSHW ZPStrBuf1
				>KAPI StrVGet

.1				>PUSHW.G pStrings
				>PUSHW STRID
				>PUSHW ZPStrBuf1
				>KAPI StrVSet

				inc pStack
				inc pStack

				rts

RT.INPUT		lda #0
				sta (ZPStrBuf1)

				stz ZPPtr1

				ldy #0					OUT.ELW
				jsr RT.OutY

.1				jsr RT.GetChar
				bcs .9

				ldy ZPPtr1

				cmp #C.SPACE
				bcs .7

				cmp #C.CR
				bne .2

				lda #0
				sta (ZPStrBuf1),y

				ldy #4					OUT.CRLF
				jsr RT.OutY

				bra .8

.2				cmp #C.BS
				bne .1

				cpy #0
				beq .1

				ldy #6					OUT.BSSPBS
				jsr RT.OutY

				dec ZPPtr1
				bra .1

.7				cpy #255
				bcs .1

				sta (ZPStrBuf1),y
				>LIBC PutChar
				inc ZPPtr1
				bra .1

.8				ldy #2					OUT.DLW
				jsr RT.OutY

				>PUSHWZ					RVALUE in buffer
				clc

.9				rts

				inc pStack

				jsr RT.StrPull

				sta (ZPStrBuf1),y

.80				>PUSHWZ					RVALUE in buffer

				rts

				adc #0
