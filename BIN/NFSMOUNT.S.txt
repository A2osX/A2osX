  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF bin/nfsmount
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/a2osx.api.i
				.INB inc/eth.i
				.INB inc/ip.i
				.INB inc/net.skt.i
				.INB inc/net.telnet.i
				.INB inc/net.db.i
*--------------------------------------
TIMEOUT.MAX		.EQ 20
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
ZPFrameBase		.BS 2
ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #S.PS.SIG.T1SEC
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG
				.DA	CS.QUIT
*--------------------------------------
L.MSG.USAGE		.DA MSG.USAGE
				.DA 0
*--------------------------------------
CS.INIT			clc
				rts
*--------------------------------------
CS.RUN			>LDYAI UDP.PORT.PMAP
				>STYA.G SA.REMOTE+S.SA.IN.PORT

.99				>LDYA L.MSG.USAGE
				>LIBC PutS
				lda #E.SYN

				sec
				rts
*--------------------------------------
CS.SIG			rts
*--------------------------------------
CS.QUIT			clc
				rts
*--------------------------------------
CS.END
MSG.USAGE		.CZ "Usage : NFSMOUNT [HOST | IP]/export /mountvol"
*--------------------------------------
SA.LOCAL		.DA AF_INET
				.DA #0,#0,#0,#0			any address
				.DA 0					Dyn port
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0

pAI				.BS 2
SA.REMOTE		.BS S.SA.IN

DS.END			.ED
