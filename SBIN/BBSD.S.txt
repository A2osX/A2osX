NEW
  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF sbin/bbsd
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
				.INB inc/mli.i
				.INB inc/mli.e.i
				.INB inc/eth.i
				.INB inc/ip.i
				.INB inc/net.skt.i
*--------------------------------------
TIMEOUT.MAX		.EQ 20					20 sec.
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
ZPSktPtr		.BS 2

ZPMsgFile		.BS 2
ZPStartupFile	.BS 2

pFile			.BS 2
pLineBuf		.BS 2
pLine			.BS 2

pMsgFile		.BS 2
pStartupFile	.BS 2

hSrvSocket		.BS 2
hCltSocket		.BS 2

TimeOut			.BS 1

ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #S.PS.SIG.T1SEC
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
* Relocation Table
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG
				.DA	CS.QUIT
*--------------------------------------
L.ETCBBSDCONF	.DA ETCBBSDCONF
L.KEYWORDS.CONF	.DA KEYWORDS.CONF
J.KEYWORDS.CONF	.DA CS.RUN.CONF.TCPLISTEN
				.DA CS.RUN.CONF.COMLISTEN
				.DA CS.RUN.CONF.MSGFILE
				.DA CS.RUN.CONF.STARTUPFILE
L.MSG.INITOK	.DA MSG.INITOK
L.MSG.SKTERR	.DA MSG.SKTERR
L.MSG.INCOMING	.DA MSG.INCOMING
L.SA.LOCAL		.DA SA.LOCAL
L.NOD.Template	.DA NOD.Template
L.CMD.Template	.DA CMD.Template
				.DA 0
*--------------------------------------
CS.INIT			clc
CS.INIT.RTS		rts
*--------------------------------------
CS.RUN			jsr CS.RUN.CONF
				bcs CS.INIT.RTS

.1				>SS
				>PUSHWI AF_INET
				>PUSHWI SOCK_STREAM
				>PUSHWZ
				>LIBC Socket
				>SR
				bcs .9

				>STYA hSrvSocket

				>SS
				>PUSHW hSrvSocket
				>PUSHW L.SA.LOCAL
				>PUSHWI S.SA.IN
				>LIBC Bind
				>SR
				bcs .9

				>SS
				>PUSHW hSrvSocket
				>PUSHWI 16				backlog
				>LIBC Listen
				>SR
				bcs .9

				>LDYA L.MSG.INITOK
				>LIBC PutS
				bcc CS.RUN.LOOP

.9				pha
				>LDYA L.MSG.SKTERR
				>LIBC PutS
				pla
				sec
				rts
*--------------------------------------
CS.RUN.LOOP		>SLEEP

				>SS
				>PUSHW hSrvSocket
				>PUSHEA.G SA.CLNT
				>PUSHWI S.SA.IN
				>LIBC Accept
				>SR
				bcs CS.RUN.LOOP

				>STYA hCltSocket

				jsr CS.RUN.CLIENT

				bcc CS.RUN.LOOP

				>SS
				>PUSHW hCltSocket
				>PUSHWI 0				how
				>LIBC Shutdown
				>SR
				bra CS.RUN.LOOP
*--------------------------------------
CS.RUN.CLIENT	>SLEEP					give some time for TCPIP SYN/ACK

				>SS
				>PUSHW L.MSG.INCOMING

				>PUSHB hCltSocket

				ldx #3
				ldy #SA.CLNT+S.SA.IN.ADDR

.1				>PUSHB (pData),y
				iny
				dex
				bpl .1

				>PUSHW.G SA.CLNT+S.SA.IN.PORT

				>PUSHBI 7
				>LIBC PrintF
				>SR

				>SS
				>PUSHEA.G NodBuf
				>PUSHW L.NOD.Template
				>PUSHB hCltSocket
				>PUSHBI 1
				>LIBC SPrintF
				>SR

CS.RUN.CLIENT1	>SS
				>PUSHEA.G NodBuf
				>PUSHWZ
				>PUSHW hCltSocket
				>LIBC MKNod
				>SR
				bcs .9

				>SS
				>PUSHEA.G CmdBuf
				>PUSHW L.CMD.Template
				>PUSHEA.G NodBuf
				>PUSHBI 2
				>LIBC SPrintF
				>SR
				bcs .9

				>SS
				>PUSHEA.G CmdBuf
				>PUSHWI 0
				>LIBC ExecL
				>SR

.9				rts
*--------------------------------------
CS.SIG			lda TimeOut
				beq .9

				dec TimeOut

.9				rts
*--------------------------------------
CS.QUIT			ldy hSrvSocket
				beq .8

				lda #0

				>LIBC Shutdown

.8				clc

CS.QUIT.RTS		rts
*--------------------------------------
CS.RUN.CONF		>SS
				>PUSHW L.ETCBBSDCONF
				>PUSHBI	O.RDONLY+O.TEXT
				>PUSHBI S.FI.T.TXT
				>PUSHWZ
				>LIBC FOpen
				>SR
				bcs CS.QUIT.RTS

				>STYA pFile

				>LDYAI 256
				>LIBC Malloc
				bcs .97

				>STYA pLineBuf

.1				>SS
				>PUSHW pLineBuf
				>PUSHWI 256
				>PUSHW pFile
				>LIBC FGetS
				>SR
				bcs .8

				tya
				beq .1

				lda (pLineBuf)
				cmp #'#'
				beq .1

				>LDYA L.KEYWORDS.CONF
				>STYA R1

				ldx #0

.2				ldy #0

.3				lda (R1),y
				cmp	(pLineBuf),y
				bne .4

				iny
				cmp #C.SPACE
				bne .3

				jsr CS.RUN.CONF.JMP
				bcc .1

				bra .97

.4				inx
				inx

				ldy #0

.5				iny
				lda (R1),y
				cmp #C.SPACE
				bne .5

				tya
				sec
				adc R1
				sta R1
				bcc .6
				inc R1+1

.6				lda	(R1)
				bne .2

.96				lda #E.SYN
				sec
				.HS 90					BCC
.8				clc

.97				php
				pha

				>LDYA pLineBuf
				beq .98

				>LIBC Free

.98				>LDYA pFile
				beq .99

				>LIBC FClose

.99				pla
				plp
				rts
*--------------------------------------
CS.RUN.CONF.JMP	tya
				clc
				adc pLineBuf
				sta R1
				tay
				lda #0
				adc pLineBuf+1
				sta R1+1

				jmp (J.KEYWORDS.CONF,x)
*--------------------------------------
CS.RUN.CONF.TCPLISTEN
				>LIBC AToI
				bcs .9

				>STYA SA.LOCAL+S.SA.IN.PORT

.9				rts
*--------------------------------------
CS.RUN.CONF.COMLISTEN

.9				clc
				rts
*--------------------------------------
CS.RUN.CONF.MSGFILE
				>SS
				>PUSHYA
				>PUSHWZ					Allocate
				>LIBC RealPath
				>SR
				bcs .9

				>STYA ZPMsgFile

.9				rts
*--------------------------------------
CS.RUN.CONF.STARTUPFILE
				>SS
				>PUSHYA
				>PUSHWZ					Allocate
				>LIBC RealPath
				>SR
				bcs .9

				>STYA ZPStartupFile

.9				rts
*--------------------------------------
CS.END			.EQ *
*--------------------------------------
ETCBBSDCONF		.AZ "/etc/bbsd.conf"
*--------------------------------------
KEYWORDS.CONF	.AS "tcplisten "
				.AS "comlisten "
				.AS "msgfile "
				.AS "startupfile "
				.DA #0
*--------------------------------------
MSG.INITOK		.AZ "BBSD:Init Ok, Listening."
MSG.SKTERR		.AZ "BBSD:Listen Error."
MSG.INCOMING	.CZ "BBSD:Incoming Connection [SKT=%h] From : %d.%d.%d.%d\r\n"
*--------------------------------------
NOD.Template	.AZ "/dev/tty%h"
CMD.Template	.AZ "/sbin/getty -E %s %s"
*--------------------------------------
SA.LOCAL		.DA AF_INET				S.SOCKADDR.AF
				.DA #0,#0,#0,#0			S.SA.IN.ADDR
				.DA 6502
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0

SA.CLNT			.BS S.SA.IN
NodBuf			.BS 11
CmdBuf			.BS 65
DS.END			.ED
*--------------------------------------

SAVE usr/src/sbin/bbsd.s
