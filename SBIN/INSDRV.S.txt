NEW
  AUTO 3,1
				.LIST OFF
				.OP	65C02
				.OR	$2000
				.TF sbin/insdrv
*--------------------------------------
				.INB inc/macros.i
				.INB inc/a2osx.i
*--------------------------------------
				.DUMMY
				.OR ZPBIN
ZS.START
ZPPTR1			.BS 2
ZPPTR2			.BS 2
ZPPTR3			.BS 2

ZS.END			.ED
*--------------------------------------
*			File Header (16 Bytes)
*--------------------------------------
CS.START		cld
				jmp (.1,x)
				.DA #$61				6502,Level 1 (65c02)
				.DA #2					BIN Layout Version 2
				.DA #0
				.DA #0
				.DA CS.END
				.DA ID.END
				.DA DS.END
				.DA #64					Stack Size
				.DA #ZS.END-ZS.START	Zero Page Size
*--------------------------------------
* Relocation Table
*--------------------------------------
.1				.DA CS.INIT
				.DA CS.RUN
				.DA CS.SIG		
				.DA	CS.QUIT
L.MSG.USAGE		.DA MSG.USAGE
L.MSG.LOAD		.DA MSG.LOAD
				.DA 0
*--------------------------------------
CS.RUN			ldy #S.PS.ARGC
				jsr A2osX.GetPSy
				cmp #1
				bcc .9
				
				>SS
				>PUSHW L.MSG.LOAD
				lda #1					get PTR to ARG[1] (Driver name)
				>LIBC ArgV
				>PUSHYA
				>PUSHBI 2
				>LIBC PrintF
				>SR
				bcs .99
				
				lda #1					get PTR to ARG[1] (Driver name)
				>LIBC ArgV

				>LIBC LoadDrv		YA = PTR to ARG[1...n]
				bcs .99

				lda #0					success, but....
				sec						we do not want to stay in memory
				rts

.9				>LDYA L.MSG.USAGE
				>LIBC PutS
				lda #E.SYN			tell TSKMGR that all done ok, but
				sec						we do not want to stay in memory
.99				rts
*--------------------------------------
CS.INIT
*--------------------------------------
CS.QUIT			clc
				rts
*--------------------------------------
CS.SIG			sec
				rts
*--------------------------------------
CS.END			.EQ *
*--------------------------------------
MSG.USAGE		.CZ "Usage: insdrv file.drv <args>"
MSG.LOAD		.CZ "INSDRV:Loading %s...\r\n"
*--------------------------------------
ID.END			.EQ *
*--------------------------------------
				.DUMMY
				.OR 0
DS.END			.ED
*--------------------------------------

SAVE usr/src/sbin/insdrv.s
ASM
